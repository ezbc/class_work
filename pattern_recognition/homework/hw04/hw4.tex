            \bm{q_1} = \bm{a}_1 \qquad \bm{q_2} =
            \frac{\bm{a}_2^\prime}{\|\bm{a}_2^\prime\|}, \qquad
            \bm{a}_2^\prime = \bm{a}_2 - (\bm{q}_1^T \bm{a}_2)\bm{q}_1
        \end{equation*}

        \noindent which gives us

        \begin{equation*}
            \bm{q}_1 = \begin{matrix} 1\\0\\0\\ \end{matrix}
